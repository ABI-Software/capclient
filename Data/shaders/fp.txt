uniform sampler2D texture0;
uniform sampler1D texture1;
void main()
{
	vec3 texColour = vec3(texture2D(texture0, vec2(gl_TexCoord[0])));
	//texColour *= 5.0;

	vec4 brightnessAndContrast = texture1D(texture1,0.0);	
	
	float contrast = 1.0/(1.0 - 1.5 * (brightnessAndContrast.y - 0.5));
	float brightness = 1.0 - brightnessAndContrast.x;
	vec3 colour =  contrast * (texColour - brightness) + 0.5;
	gl_FragColor = vec4(colour, 1);
}
